-- capstone DB 사용
USE capstone;


-- 테이블 확인
SHOW TABLES;


-- 관리자 계정 없을 때 생성
INSERT IGNORE INTO users (username, password, role)
VALUES ('admin', '1234', 'ADMIN');


-- 관리자 계정 확인
SELECT * FROM users;


-- 관리자 비밀번호 1234로 초기화
UPDATE users
SET password = '1234', role = 'ADMIN'
WHERE username = 'admin';


-- 시험 폴더까지 전체 초기화, admin 계정 유지
SET FOREIGN_KEY_CHECKS = 0;

TRUNCATE TABLE objective_submission;
TRUNCATE TABLE objective_question;
TRUNCATE TABLE submission;
TRUNCATE TABLE test_case;
TRUNCATE TABLE exam_record;
TRUNCATE TABLE exam;
TRUNCATE TABLE exam_category;

-- admin 계정은 남기고 나머지 계정 삭제
DELETE FROM users
WHERE username <> 'admin';

-- admin만 남긴 상태에서 다음 학생 ID를 다시 2부터 시작하고 싶을 때
ALTER TABLE users AUTO_INCREMENT = 2;

SET FOREIGN_KEY_CHECKS = 1;

-- 시험 폴더 확인
SELECT * FROM exam_category ORDER BY id;


-- 코딩 문제 확인
SELECT * FROM exam ORDER BY id;


-- 테스트케이스 확인
SELECT * FROM test_case ORDER BY id;


-- 코딩 제출 확인
SELECT * FROM submission ORDER BY id;


-- 객관식 문제 확인
SELECT * FROM objective_question ORDER BY id;


-- 객관식 제출 확인
SELECT * FROM objective_submission ORDER BY id;


-- 학생별 코딩 점수 확인
SELECT
    student_id,
    student_name,
    SUM(earned_point) AS coding_score
FROM submission
GROUP BY student_id, student_name
ORDER BY coding_score DESC;


-- 학생별 객관식 점수 확인
SELECT
    student_id,
    student_name,
    SUM(earned_point) AS objective_score
FROM objective_submission
GROUP BY student_id, student_name
ORDER BY objective_score DESC;


-- 문제별 제출 상태 확인
SELECT
    exam_id,
    status,
    COUNT(*) AS count
FROM submission
GROUP BY exam_id, status
ORDER BY exam_id;
